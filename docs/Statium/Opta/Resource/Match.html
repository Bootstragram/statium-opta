def uri(auth_key, lang = 'en')
  specific_options = {
    tmcl: @tournament_calendar_uuid,
    live: 'yes',
    lineups: 'yes',
    _pgSz: @page_size,
  }
  build_uri('match', auth_key, nil, common_url_options(lang).merge(specific_options))
end
